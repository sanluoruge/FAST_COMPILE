30/* Change all stepsize using the newBaseRateStepSize */



31void CFDLMFAC_ChangeStepSize(real_T newBaseRateStepSize, RT_MODEL_CFDLMFAC_T *



32  const CFDLMFAC_M)



33{



34  real_T ratio = newBaseRateStepSize / 0.00625;



35



36  /* update non-zore stepsize of periodic



37   * sample time. Stepsize of asynchronous



38   * sample time is not changed in this function */



39  CFDLMFAC_M->Timing.stepSize0 = CFDLMFAC_M->Timing.stepSize0 * ratio;



40  CFDLMFAC_M->Timing.stepSize1 = CFDLMFAC_M->Timing.stepSize1 * ratio;



41  CFDLMFAC_M->Timing.stepSize = CFDLMFAC_M->Timing.stepSize * ratio;



42}



43



44void XPCCALLCONV changeStepSize(real_T stepSize)



45{



46  /* Change all stepsize using the newBaseRateStepSize */



47  CFDLMFAC_ChangeStepSize(stepSize, CFDLMFAC_M);



48}
